static Map<T, V>? tryJson<T, V>(dynamic data, {WebRequest? rq}) {
  try {
    var res = jsonEncoder(data, rq: rq);
    return jsonDecoder(res) as Map<T, V>;
  } catch (e) {
    print(e);
    return null;
  }
}
